data = functools.reduce(one_hot, ["work_type", "smoking_status"], data)

import numpy as np
import matplotlib.pyplot as plt

def ceteris_paribus(model, observation, feature, min_value, max_value, N, plot=True):
  values = np.linspace(min_value, max_value, N)
  observations = pd.concat([observation] * N, ignore_index=True)
  observations[feature] = values
  scores = model(observations)

  if plot:
    fig, ax = plt.subplots()
    ax.set_xlabel(feature)
    ax.set_ylabel("prediction")
    ax.plot(values, scores)
    ax.plot(observation[[feature]].values[0, 0], model(observation), 'o')
  else:
    return values, scores

def ceteris_paribus_one_hot(model, observation, features, plot=True):
  N = len(features)
  values = np.identity(N)

  observations = pd.concat([observation] * N, ignore_index=True)
  observations[features] = values
  scores = model(observations)

  if plot:
    colors = ['blue' if val == 0 else "orange" for val in observation[features].values[0,:]]
    fig, ax = plt.subplots()
    ax.set_ylabel("predictions")
    ax.bar(features, scores, color=colors)
    ax.set_xticks(ax.get_xticks(), ax.get_xticklabels(), rotation=45, ha='right')
  else:
    return values, scores

def partial_dependency_profile(model, data, feature, min_value, max_value, N, plot=True):
  values = np.linspace(min_value, max_value, N)
  values_repeat = values.repeat(len(data))

  data = pd.concat([data] * N, ignore_index=True)
  data[feature] = values_repeat
  scores = model(data).reshape(N, -1).mean(axis=1)

  if plot:
    fig, ax = plt.subplots()
    ax.set_xlabel(feature)
    ax.set_ylabel("prediction")
    ax.plot(values, scores)
  else:
    return values, scores

def partial_dependency_profile_one_hot(model, data, features, plot=True):
  N = len(features)
  values = np.identity(N)
  values_repeat = values.repeat(len(data), axis=0)

  observations = pd.concat([data] * N, ignore_index=True)
  observations[features] = values_repeat
  scores = model(observations).reshape(N, -1).mean(axis=1)

  if plot:
    fig, ax = plt.subplots()
    ax.set_ylabel("predictions")
    ax.bar(features, scores)
    ax.set_xticks(ax.get_xticks(), ax.get_xticklabels(), rotation=45, ha='right')
  else:
    return values, scores
